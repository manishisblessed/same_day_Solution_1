2. Authentication

Every request requires 3 custom headers:

HeaderDescription
x-api-keyYour partner API key
x-signatureHMAC-SHA256 signature (see formula below)
x-timestampCurrent Unix timestamp in milliseconds

Signature Formula

For POST requests (with JSON body):

signature = HMAC_SHA256(api_secret, JSON.stringify(body_object) + timestamp)

For GET requests (no body):

signature = HMAC_SHA256(api_secret, "" + timestamp)
Critical: The signature must be computed on compact (minified) JSON — no extra whitespace, newlines, or indentation. The server parses and re-serializes your body before verifying. If you sign formatted JSON, the signature WILL fail.
Security Notes:
  • IP whitelisting enforced — requests from non-whitelisted IPs get 403
  • Timestamp must be within 5 minutes of server time
  • The api_secret is used only for signing — never send it in headers or params

3. Payment Flow

1 Get Billers — List available credit card billers (save product_code)
↓
2 Fetch Bill — Fetch bill using card last 4 digits + mobile (save order_id as bill_fetch_ref)
↓
3 Check Charges — See applicable charges + GST for a given amount
↓
4 Pay Bill — Process payment with bill_fetch_ref from Step 2
↓
5 Check Status — Verify transaction status (use for timeout/ambiguous cases)
Key Points:
  • No retailer_id needed — charges and wallet debit are handled at partner level
  • number field = Last 4 digits of credit card (NOT full 16 digits)
  • optional1 = Registered mobile number (mandatory for credit card billers)
  • Amount is in rupees (not paise)
  • Charges include 18% GST on the base service charge
  • Auto-refund occurs if upstream payment fails after wallet debit

6. Check Charges

POST

Check Payment Charges

/api/partner/pay2new/charges

Check applicable service charges + GST before making a payment. Charges are resolved from your partner scheme — no retailer_id required.

Request Body

FieldTypeRequiredDescription
amountIntegerYesBill payment amount in rupees

Example Request

{
  "amount": 15234
}

Response Fields

FieldTypeDescription

charges.base_chargeNumberService charge before GST
charges.gst_percentNumberGST percentage (18%)

8. Check Status

POST

Check Transaction Status

/api/partner/pay2new/bill/status

Check the status of a bill payment. Use when Pay Bill times out or you need to verify the final status.

Request Body

Provide either order_id or request_id (at least one required):

FieldTypeRequiredDescription
order_idStringOne of theseorder_id from Pay Bill success response
request_idStringis requiredrequest_id from Pay Bill response (works for failed txns too)

9. Error Codes

All error responses follow this structure:

{
  "success": false,
  "error": {
    "code": "ERROR_CODE",
    "message": "Human-readable description"
  }

10. Code Samples

Node.js — Complete Flow

const crypto = require('crypto');

const API_KEY    = 'pk_live_YOUR_API_KEY';
const API_SECRET = 'sk_live_YOUR_API_SECRET';
const BASE_URL   = 'https://api.samedaysolution.in';

function sign(body, timestamp) {
  const payload = (body ? JSON.stringify(body) : '') + timestamp;
  return crypto.createHmac('sha256', API_SECRET).update(payload).digest('hex');
}

function headers(body) {
  const timestamp = Date.now().toString();
  return {
    'Content-Type': 'application/json',
    'x-api-key': API_KEY,
    'x-signature': sign(body, timestamp),
    'x-timestamp': timestamp
  };
}

// Step 1: Get Billers
async function getBillers() {
  const res = await fetch(`${BASE_URL}/api/partner/pay2new/billers`, {
    headers: headers(null)
  });
  return res.json();
}

// Step 2: Fetch Bill
async function fetchBill(lastFourDigits, productCode, mobile) {
  const body = {
    number: lastFourDigits,
    product_code: productCode,
    customer_number: mobile,
    optional1: mobile,
    optional2: '', optional3: '', optional4: '',
    pincode: '414002'
  };
  const res = await fetch(`${BASE_URL}/api/partner/pay2new/bill/fetch`, {
    method: 'POST', headers: headers(body), body: JSON.stringify(body)
  });
  return res.json();
}

// Step 3: Check Charges
async function checkCharges(amount) {
  const body = { amount };
  const res = await fetch(`${BASE_URL}/api/partner/pay2new/charges`, {
    method: 'POST', headers: headers(body), body: JSON.stringify(body)
  });
  return res.json();
}

// Step 4: Pay Bill
async function payBill(lastFour, amount, productCode, billFetchRef, mobile) {
  const body = {
    number: lastFour, amount, product_code: productCode,
    bill_fetch_ref: billFetchRef, customer_number: mobile,
    optional1: mobile, optional2: '', optional3: '', optional4: '',
    pincode: '414002'
  };
  const res = await fetch(`${BASE_URL}/api/partner/pay2new/bill/pay`, {
    method: 'POST', headers: headers(body), body: JSON.stringify(body)
  });
  return res.json();
}

// Step 5: Check Status
async function checkStatus(orderId) {
  const body = { order_id: orderId };
  const res = await fetch(`${BASE_URL}/api/partner/pay2new/bill/status`, {
    method: 'POST', headers: headers(body), body: JSON.stringify(body)
  });
  return res.json();
}

Python

import hmac, hashlib, json, time, requests

API_KEY    = "pk_live_YOUR_API_KEY"
API_SECRET = "sk_live_YOUR_API_SECRET"
BASE_URL   = "https://api.samedaysolution.in"

def sign_and_call(method, path, body=None):
    timestamp = str(int(time.time() * 1000))
    body_str = json.dumps(body, separators=(",", ":")) if body else ""
    payload = body_str + timestamp
    signature = hmac.new(
        API_SECRET.encode(), payload.encode(), hashlib.sha256
    ).hexdigest()

    hdrs = {
        "Content-Type": "application/json",
        "x-api-key": API_KEY,
        "x-signature": signature,
        "x-timestamp": timestamp,
    }

    if method == "GET":
        return requests.get(f"{BASE_URL}{path}", headers=hdrs).json()
    return requests.post(f"{BASE_URL}{path}", data=body_str, headers=hdrs).json()

# Step 1
billers = sign_and_call("GET", "/api/partner/pay2new/billers")

# Step 2
bill = sign_and_call("POST", "/api/partner/pay2new/bill/fetch", {
    "number": "5008", "product_code": "ICICI_CREDIT_CARD",
    "customer_number": "9971969046", "optional1": "9971969046",
    "optional2": "", "optional3": "", "optional4": "", "pincode": "414002"
})

# Step 3
charges = sign_and_call("POST", "/api/partner/pay2new/charges", {"amount": 15234})

# Step 4
payment = sign_and_call("POST", "/api/partner/pay2new/bill/pay", {
    "number": "5008", "amount": 15234,
    "product_code": "ICICI_CREDIT_CARD",
    "bill_fetch_ref": bill["order_id"],
    "customer_number": "9971969046", "optional1": "9971969046",
    "optional2": "", "optional3": "", "optional4": "", "pincode": "414002"
})

# Step 5
status = sign_and_call("POST", "/api/partner/pay2new/bill/status",
    {"order_id": payment["order_id"]})
