2. Authentication

Every request requires 3 custom headers:

HeaderDescription
x-api-keyYour partner API key
x-signatureHMAC-SHA256 signature
x-timestampCurrent Unix timestamp in milliseconds

Signature Formula

For POST requests:

signature = HMAC_SHA256(api_secret, JSON.stringify(body) + timestamp)

For GET / DELETE requests:

signature = HMAC_SHA256(api_secret, "" + timestamp)
Security:
  • IP whitelisting enforced — requests from non-whitelisted IPs get 401
  • Timestamp must be within 5 minutes of server time
  • api_secret is used only for signing — never send it in headers

3. Settlement Flow

1 Get Balance — Check wallet balance
↓
2 Add & Verify Account — Penny-drop verification (₹4 charge)
↓
3 List Accounts — View verified accounts
↓
4 Get Charges — Check transfer charges for a given amount
↓
5 Transfer — Initiate settlement bank transfer
↓
6 Check Status — Track transaction status & get UTR
Key Points:
  • All transfers go to pre-verified accounts only
  • Account verification uses penny-drop (₹1 transfer) and costs ₹4
  • Transfer charges for partners: ₹0 (direct wallet debit)
  • Duplicate prevention: 2-minute cooldown per account
  • If transfer fails, amount is auto-refunded to partner wallet

5. Add & Verify Account

/api/partner/settlement/accounts

Add a bank account and verify via penny-drop (₹1 transfer). Verification charge: ₹4 (debited from wallet).

Request Body

account_holder_nameStringYesAccount holder name

{

  "contact_mobile": "9971969046"
}

Response Fields

FieldTypeDescription
successBooleanRequest success status
verifiedBooleanWhether account was verified
verification_statusStringSUCCESS | PENDING | FAILED
verified_nameStringName returned by bank
account.idStringAccount ID — use this for transfers
charge_deductedNumber₹4

Example Response

200 Account Verified

{
  "success": true,
  "verified": true,
  "verification_status": "SUCCESS",
  "verified_name": "MANISH KUMAR SHAH",

    "account_holder_name": "Manish Kumar Shah"

If verification_status is PENDING, the account will be verified shortly. Use List Accounts to check the updated status.

13. Code Samples

Node.js — Complete Flow

const crypto = require('crypto');

const API_KEY    = 'pk_live_YOUR_API_KEY';
const API_SECRET = 'sk_live_YOUR_API_SECRET';
const BASE_URL   = 'https://api.samedaysolution.in';

function getHeaders(body = null) {
  const timestamp = Date.now().toString();
  const payload = (body ? JSON.stringify(body) : '') + timestamp;
  const signature = crypto.createHmac('sha256', API_SECRET)
    .update(payload).digest('hex');
  return {
    'Content-Type': 'application/json',
    'x-api-key': API_KEY,
    'x-signature': signature,
    'x-timestamp': timestamp
  };
}

// 1. Check Balance
async function getBalance() {
  const res = await fetch(`${BASE_URL}/api/partner/settlement/balance`, {
    headers: getHeaders()
  });
  return res.json();
}

// 2. Add & Verify Account
async function addAccount(accountNumber, ifsc, holderName, mobile) {
  const body = {
    account_number: accountNumber,
    ifsc_code: ifsc,
    account_holder_name: holderName,
    contact_name: holderName,
    contact_mobile: mobile
  };
  const res = await fetch(`${BASE_URL}/api/partner/settlement/accounts`, {
    method: 'POST', headers: getHeaders(body), body: JSON.stringify(body)
  });
  return res.json();
}

// 3. List Accounts
async function listAccounts() {
  const res = await fetch(`${BASE_URL}/api/partner/settlement/accounts`, {
    headers: getHeaders()
  });
  return res.json();
}

// 4. Transfer
async function transfer(accountId, amount, mode = 'IMPS') {
  const body = { account_id: accountId, amount, mode };
  const res = await fetch(`${BASE_URL}/api/partner/settlement/transfer`, {
    method: 'POST', headers: getHeaders(body), body: JSON.stringify(body)
  });
  return res.json();
}

// 5. Check Status
async function checkStatus(referenceId) {
  const res = await fetch(
    `${BASE_URL}/api/partner/settlement/status?reference_id=${referenceId}`,
    { headers: getHeaders() }
  );
  return res.json();
}

Python

import hmac, hashlib, json, time, requests

API_KEY    = "pk_live_YOUR_API_KEY"
API_SECRET = "sk_live_YOUR_API_SECRET"
BASE_URL   = "https://api.samedaysolution.in"

def call_api(method, path, body=None, params=None):
    timestamp = str(int(time.time() * 1000))
    body_str = json.dumps(body, separators=(",", ":")) if body else ""
    payload = body_str + timestamp
    signature = hmac.new(
        API_SECRET.encode(), payload.encode(), hashlib.sha256
    ).hexdigest()

    hdrs = {
        "Content-Type": "application/json",
        "x-api-key": API_KEY,
        "x-signature": signature,
        "x-timestamp": timestamp,
    }

    url = f"{BASE_URL}{path}"
    if method == "GET":
        return requests.get(url, headers=hdrs, params=params).json()
    elif method == "DELETE":
        return requests.delete(url, headers=hdrs, params=params).json()
    return requests.post(url, data=body_str, headers=hdrs).json()

# Check Balance
balance = call_api("GET", "/api/partner/settlement/balance")

# Add Account
account = call_api("POST", "/api/partner/settlement/accounts", {
    "account_number": "50100104420821",
    "ifsc_code": "HDFC0003756",
    "account_holder_name": "Manish Kumar Shah",
    "contact_mobile": "9971969046"
})

# Transfer
txn = call_api("POST", "/api/partner/settlement/transfer", {
    "account_id": account["account"]["id"],
    "amount": 5000,
    "mode": "IMPS"
})

# Check Status
status = call_api("GET", "/api/partner/settlement/status",
    params={"reference_id": txn["transaction"]["reference_id"]})
