1. Authentication

Every authenticated request requires 3 custom headers:

HeaderDescription
x-api-keyYour public API key (starts with pk_live_)
x-signatureHMAC-SHA256 signature (see formula below)
x-timestampCurrent Unix timestamp in milliseconds

Signature Formula

For POST requests (with JSON body):

signature = HMAC_SHA256(api_secret, JSON.stringify(body) + timestamp)

For GET requests (no body):

signature = HMAC_SHA256(api_secret, "" + timestamp)
Important: The api_secret (starts with sk_live_) is used only for signing — never send it as a header or query parameter.

Quick Start

  1. Receive your api_key and api_secret from the Same Day Solution admin team.
  2. Whitelist your server IP(s) — contact admin.
  3. Import the Postman collection (attached) and set your keys in Collection Variables.
  4. Make a Health Check call to verify connectivity.

3. Security & Rate Limits

  • IP Whitelisting: Only whitelisted IPs can access authenticated endpoints. Contact admin to add IPs.
  • Timestamp Tolerance: Requests must arrive within 5 minutes (300 seconds) of the x-timestamp value.
  • HTTPS: Required in production. Plain HTTP is rejected.
  • Rate Limit: 100 requests/minute per API key. Exceeding returns 429 with Retry-After header.

9. Error Codes

All error responses follow this structure:

{
  "success": false,
  "error": {
    "code": "ERROR_CODE",
    "message": "Human-readable description"
  }
}
HTTP CodeError CodeDescription
400BAD_REQUESTMissing or invalid parameters (e.g. date_from required)
401UNAUTHORIZEDInvalid API key, bad signature, expired timestamp, or IP not whitelisted
404NOT_FOUNDResource not found (e.g. invalid export job_id)
429RATE_LIMIT_EXCEEDEDToo many requests — check Retry-After header for wait time
Common 401 causes:
  • API key not set or incorrect
  • Signature computed incorrectly (double-check JSON.stringify order)
  • Timestamp older than 5 minutes
  • Server IP not whitelisted — if using Postman, disable cloud agent (Settings → General → "Use Postman's cloud agent" → OFF)

10. Code Samples

Node.js (JavaScript)

const crypto = require('crypto');

const API_KEY  = 'pk_live_YOUR_API_KEY';
const API_SECRET = 'sk_live_YOUR_API_SECRET';
const BASE_URL = 'https://api.samedaysolution.in';

async function getTransactions(dateFrom, dateTo, options = {}) {
  const body = {
    date_from: dateFrom,
    date_to: dateTo,
    status: options.status || null,
    terminal_id: options.terminalId || null,
    payment_mode: options.paymentMode || null,
    page: options.page || 1,
    page_size: options.pageSize || 50
  };

  const timestamp = Date.now().toString();
  const payload = JSON.stringify(body) + timestamp;
  const signature = crypto
    .createHmac('sha256', API_SECRET)
    .update(payload)
    .digest('hex');

  const res = await fetch(`${BASE_URL}/api/partner/pos-transactions`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'x-api-key': API_KEY,
      'x-signature': signature,
      'x-timestamp': timestamp
    },
    body: JSON.stringify(body)
  });

  return res.json();
}

// Usage
const data = await getTransactions(
  '2026-02-01T00:00:00.000Z',
  '2026-02-28T23:59:59.999Z',
  { status: 'CAPTURED', page: 1 }
);
console.log(data);

Python

import hmac, hashlib, json, time, requests

API_KEY    = "pk_live_YOUR_API_KEY"
API_SECRET = "sk_live_YOUR_API_SECRET"
BASE_URL   = "https://api.samedaysolution.in"

def get_transactions(date_from, date_to, **kwargs):
    body = {
        "date_from": date_from,
        "date_to": date_to,
        "status": kwargs.get("status"),
        "terminal_id": kwargs.get("terminal_id"),
        "payment_mode": kwargs.get("payment_mode"),
        "page": kwargs.get("page", 1),
        "page_size": kwargs.get("page_size", 50),
    }

    timestamp = str(int(time.time() * 1000))
    payload = json.dumps(body, separators=(",", ":")) + timestamp
    signature = hmac.new(
        API_SECRET.encode(), payload.encode(), hashlib.sha256
    ).hexdigest()

    resp = requests.post(
        f"{BASE_URL}/api/partner/pos-transactions",
        json=body,
        headers={
            "Content-Type": "application/json",
            "x-api-key": API_KEY,
            "x-signature": signature,
            "x-timestamp": timestamp,
        },
    )
    return resp.json()

# Usage
data = get_transactions(
    "2026-02-01T00:00:00.000Z",
    "2026-02-28T23:59:59.999Z",
    status="CAPTURED",
)
print(data)

cURL

# Step 1: Generate signature (using Node.js one-liner)
TIMESTAMP=$(date +%s%3N)
BODY='{"date_from":"2026-02-01T00:00:00.000Z","date_to":"2026-02-28T23:59:59.999Z","status":null,"terminal_id":null,"payment_mode":null,"page":1,"page_size":50}'
SIGNATURE=$(echo -n "${BODY}${TIMESTAMP}" | openssl dgst -sha256 -hmac "sk_live_YOUR_SECRET" | awk '{print $2}')

# Step 2: Make the request
curl -X POST https://api.samedaysolution.in/api/partner/pos-transactions \
  -H "Content-Type: application/json" \
  -H "x-api-key: pk_live_YOUR_KEY" \
  -H "x-signature: ${SIGNATURE}" \
  -H "x-timestamp: ${TIMESTAMP}" \
  -d "${BODY}"
